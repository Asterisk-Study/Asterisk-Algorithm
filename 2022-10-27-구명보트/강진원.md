```js
function solution(people, limit) {
  let answer = 0;

  for (let i = 0; i < people.length; i++) {
    if (people[i] < limit) {
      for (let j = i + 1; j < people.length; j++) {
        if (people[i] + people[j] <= limit) {
          people.splice(j, 1);
          people.splice(i, 1);
          answer++;
        }
      }
    }
  }
  console.log(answer + people.length);
}
solution([70, 50, 80, 50, 150], 100);
```
