처음 푼 코드
```js
function solution(people, limit) {
    let count = 0;
    people.sort((a,b)=>b-a);
    
    while(people.length!==0){
        if(people[0] + people[people.length-1] > limit){
            people.shift();
            count++;
        }else{
            people.pop();
            people.splice(0,1);
            count++;
        }
    }
    
    return count;
}
```

다른 풀이 방법

function solution(people, limit) {
     let count = 0;
     people = people.sort((a,b)=>b-a);
    
     for(var i=0, j=people.length-1 ; i<=j; i++){
         if(people[i]+people[j] <= limit) j--,count++;
         else count++;
     }
        
     return count;

}
