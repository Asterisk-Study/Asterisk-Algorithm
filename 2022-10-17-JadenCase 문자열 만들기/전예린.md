function solution(s) {
    let answer = '';
    s = s.toLowerCase();
    let arr = s.split(' ');
    console.log(arr);
    for(let i=0;i<arr.length;i++){
        if(isNaN(arr[i][0])&&arr[i].length!==0){
            arr[i] = arr[i][0].toUpperCase() + arr[i].slice(1);
        }
    }
    answer = arr.join(' ');
  
    return answer;
}
